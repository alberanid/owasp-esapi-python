213  # ESAPI Validation
 
214  #
 
215  # The ESAPI validator does many security checks on input, including  
216  # canonicalization and whitelist validation. Note that all of these validation 
217  # rules are applied *after* canonicalization. Double-encoded characters  
218  # (even with different encodings involved, are never allowed.
 
219  #
 
220  # To use:
 
221  #
 
222  # First set up a pattern below. You can choose any name you want, prefixed by  
223  # the word "Validation_". It is a good idea to put regex strings in raw  
224  # triple-quoted strings. 
225  # For example:
 
226  #   Validator_Email = r"""^[A-Za-z0-9._%-]+@[A-Za-z0-9.-]+\.[a-zA-Z]{2,4}$"""
 
227  # 
 
228  # Then you can validate in your code against the pattern like this:
 
229  #   ESAPI.validator().get_valid_input( "Context", input, "Email", 100, False )
 
230  #   ESAPI.validator().is_valid_input( "Context", input, "Email", 100, False )
 
231  #

 
232  Validator_Email = r"""^[A-Za-z0-9._%-]+@[A-Za-z0-9.-]+\.[a-zA-Z]{2,4}$""" 
233  Validator_IPAddress = r"""^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$""" 
234  # This regex modified slightly from Java version: ? just before $ has been removed
 
235  Validator_URL = r"""^(ht|f)tp(s?)://[0-9a-zA-Z]([-.\w]*[0-9a-zA-Z])*(:(0-9)*)*(/?)([a-zA-Z0-9-\.\?,:'/\\\+=&amp;%\$#_]*)$""" 
236  Validator_CreditCard = r"""^(\d{4}[- ]?){3}\d{4}$""" 
237  Validator_SSN = r"""^(?!000)([0-6]\d{2}|7([0-6]\d|7[012]))([ -]?)(?!00)\d\d\3(?!0000)\d{4}$"""
